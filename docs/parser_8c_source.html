872  }

874  char line[MAX_LINE_LENGTH];
875  while (fgets(line, MAX_LINE_LENGTH, file)) {
876 
877  // Parse the production rule to remove the arrow
878  char* arrow = strstr(line, "===>");
879  if (!arrow) continue;
880  *arrow = '\0';
881  char* lhs = line;
882 
883  // Parse the rule to remove leading and trailing spaces
884  while (isspace(*lhs))
885  lhs++;
886  char* end = lhs + strlen(lhs) - 1;
887  while (end > lhs && isspace(*end))
888  end--;
889  *(end + 1) = '\0';
890 
891  // Check the LHS for non-terminal
892  if (*lhs != '<' || lhs[strlen(lhs)-1] != '>') {
893  fprintf(stderr, "Invalid non-terminal: %s\n", lhs);
